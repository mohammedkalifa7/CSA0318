#include <stdio.h>
#include <stdlib.h>
struct Node {
    int data;
 struct Node *next;
};
struct Node *top = NULL;
void push(int val) {
    struct Node *newnode = (struct Node *)malloc(sizeof(struct Node));
    newnode->data = val;
    newnode->next = top;
    top = newnode;
    printf("%d pushed to stack\n", val);
}
int main() {
    int val;
    printf("Enter value to push: ");
    scanf("%d", &val);
    push(val);
    return 0;
}
