#include <stdio.h>
#include <stdlib.h>
struct Node {
    int data;
    struct Node *next;
};

struct Node *top = NULL;
void push(int val) {
    struct Node *newnode = (struct Node *)malloc(sizeof(struct Node));
    newnode->data = val;
    newnode->next = top;
    top = newnode;
}
void pop() {
    if (top == NULL)
        printf("Stack Underflow\n");
    else {
        struct Node *temp = top;
        printf("Popped element: %d\n", temp->data);
        top = top->next;
        free(temp);
    }
}
int main() {
    push(10);
    push(20);
    push(30);
    pop();
    return 0;
}
